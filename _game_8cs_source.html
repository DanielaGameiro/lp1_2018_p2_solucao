124  public void Play()
125  {
126  // Chaves (nome) da informação estatística sobre o jogo
127  string turnKey = "Turn";
128  string totalZombiesKey = "Total Zombies";
129  string totalHumansKey = "Total Humans";
130  string playerZombiesKey = "Player Zombies";
131  string playerHumansKey = "Player Humans";
132 
133  // Valores da informação estatística sobre o jogo
134  int totalZombiesValue = (int)options.Zombies;
135  int totalHumansValue = (int)options.Humans;
136  int playerZombiesValue = (int)options.PlayerZombies;
137  int playerHumansValue = (int)options.PlayerHumans;
138 
139  // Criar um dicionário com as chaves-valores da informação
140  // estatística sobre o jogo
141  IDictionary<string, int> info = new Dictionary<string, int>()
142  {
143  { turnKey, 0 },
144  { totalZombiesKey, totalZombiesValue },
145  { totalHumansKey, totalHumansValue },
146  { playerZombiesKey, playerZombiesValue },
147  { playerHumansKey, playerHumansValue }
148  };
149 
150  // Inicializar a interface gráfica tendo em conta o tamanho do
151  // mundo de jogo
152  Program.UI.Initialize(world.XDim, world.YDim);
153 
154  // Inicializar renderização
155  Program.UI.RenderStart();
156 
157  // Renderizar mundo de jogo pela primeira vez
158  Program.UI.RenderWorld(world);
159 
160  // Renderizar informação estatística sobre o jogo pela primeira vez
161  Program.UI.RenderInfo(info);
162 
163  // Game loop:
164  // - Ciclo principal do jogo, cada iteração corresponde a um turno
165  // - Ciclo continua enquanto não tiver sido atingindo o número
166  // máximo de turnos e enquanto existirem humanos no jogo
167  for (int i = 0; i < options.Turns && totalHumansValue > 0; i++)
168  {
169  // Atualizar turno no dicionário que contém a info estatística
170  info[turnKey] = i + 1;
171 
172  // Atualizar a informação estatística na UI
173  Program.UI.RenderInfo(info);
174 
175  // Mostrar mensagem com indicação de novo turno
176  Program.UI.RenderMessage($"Starting turn {i + 1}");
177 
178  // Embaralhar array de agentes
179  Shuffle();
180 
181  // Percorrer array de agentes de modo a que eles possam
182  // realizar a sua ação no turno atual
183  foreach (Agent agent in agents)
184  {
185  // Variável que indica se houve uma alteração na composição
186  // da população (isto é, se um humano se transformou em
187  // zombie)
188  bool changePopulation;
189 
190  // Mensagem devolvida pelo agente após ter realizado a sua
191  // ação neste turno
192  string agentMessage;
193 
194  // Se for necessário esperar entre a ação de cada agente,
195  // então vamos fazê-lo agora
196  if (agentActionDelay > 0) Thread.Sleep(agentActionDelay);
197 
198  // Pedir ao agente atual para realizar a sua ação
199  agent.PlayTurn(out changePopulation, out agentMessage);
200 
201  // Houve uma alteração na população devido à ação do agente?
202  if (changePopulation)
203  {
204  // Se sim, recontar número de agentes...
205  ReCountAgents(
206  out totalZombiesValue, out totalHumansValue,
207  out playerZombiesValue, out playerHumansValue);
208  // ...atualizar dicionário com informação estatística...
209  info[totalZombiesKey] = totalZombiesValue;
210  info[totalHumansKey] = totalHumansValue;
211  info[playerZombiesKey] = playerZombiesValue;
212  info[playerHumansKey] = playerHumansValue;
213  // ...e atualizar UI com essa nova informação
214  Program.UI.RenderInfo(info);
215  }
216 
217  // Mostrar mensagem devolvida pelo agente relativa à sua
218  // ação
219  Program.UI.RenderMessage(agentMessage);
220 
221  // Atualizar mundo de simulação na UI após a ação do agente
222  // atual
223  Program.UI.RenderWorld(world);
224  }
225 
226  // Se for necessário esperar entre turnos, então vamos fazê-lo
227  // agora
228  if (turnActionDelay > 0) Thread.Sleep(turnActionDelay);
229  }
230 
231  // Se chegarmos aqui significa que o jogo terminou, sendo
232  // necessário mostrar a mensagem de vitória adequada
233  if (totalHumansValue > 0)
234  // Humanos sobreviveram!
235  Program.UI.RenderMessage(
236  $"Humans survived after {options.Turns} turns!");
237  else
238  // Zombies infetaram todos os humanos!
239  Program.UI.RenderMessage(
240  $"No humans left! Zombies won after {info[turnKey]} turns!");
241 
242  // Finalizar UI/renderização
243  Program.UI.RenderFinish();
244  }

252  private void NewAgent(AgentKind kind, AgentMovement movement, int id)
253  {

271

273  agents[id] = agent;

283  private void Shuffle()
284  {

287  Agent aux;
288  int j = rand.Next(i + 1);

291  agents[i] = aux;

312  private void ReCountAgents(
313  out int totalZombies, out int totalHumans,
314  out int playerZombies, out int playerHumans)
315  {
316  // Valores iniciais
317  totalZombies = 0;
318  totalHumans = 0;
319  playerZombies = 0;
320  playerHumans = 0;
321 
322  // Percorrer array de agentes
323  foreach (Agent ag in agents)
324  {
325  // Agente é zombie ou humano?
326  if (ag.Kind == AgentKind.Zombie)
327  {
328  // Agente é zombie, incrementar número total de zombies
329  totalZombies++;
330 
331  // Zombie é controlado pelo jogador?
332  if (ag.Movement == AgentMovement.Player)
333  {
334  // Se sim, incrementar número de zombies controlados
335  // pelo jogador
336  playerZombies++;
337  }
338  }
339  else
340  {
341  // Agente é humano, incrementar número total de humanos
342  totalHumans++;
343 
344  // Humano é controlado pelo jogador?
345  if (ag.Movement == AgentMovement.Player)
346  {
347  // Se sim, incrementar número de humanos controlados
348  // pelo jogador
349  playerHumans++;
350  }
351  }
352  }
353  }
