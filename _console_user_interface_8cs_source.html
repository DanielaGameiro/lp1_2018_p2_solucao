361

449  public void RenderMessage(string message)
450  {
451  // Variável onde guardar a última mensagem, que será mostrada com
452  // cores diferentes
453  string lastMsg = null;
454 
455  // Um string builder, que permite construir strings de forma
456  // mais eficiente (comparativamente a concatenar strings)
457  // Vai ser usado para construir todas as mensagens a imprimir de
458  // uma só vez
459  StringBuilder sb = new StringBuilder();
460 
461  // Se a fila de mensagens estiver cheia...
462  if (messageQueue.Count == messagesMaxNum)
463  {
464  // ...remover mensagem mais antiga da fila
465  messageQueue.Dequeue();
466  }
467 
468  // O tamanho da mensagem é superior ao máximo?
469  if (message.Length < messagesMaxLength)
470  {
471  // Se não for, colocar espaços no fim de modo a que a string
472  // fique com o tamanho máximo
473  // Isto permite mostrar a cor de fundo das mensagens de forma
474  // consistente
475  message = message
476  + BlankString(messagesMaxLength - message.Length);
477  }
478  else
479  {
480  // Se for, encurtar a mensagem de modo a que fique com o
481  // tamanho máximo
482  message = message.Substring(0, messagesMaxLength);
483  }
484 
485  // Colocar mensagem na fila
486  messageQueue.Enqueue(message);
487 
488  // Construir string contendo as mensagens a mostrar
489  foreach (string msg in messageQueue)
490  {
491  // Construir mensagem atual, com espaçamento à esquerda,
492  // marcador de início de mensagem, a mensagem propriamente dita
493  // e a nova linha ("\n" em Linux/Mac, "\r\n" em Windows)
494  lastMsg =
495  $"{BlankString(posMessagesLeft)}{msgBullet}{msg}{Environment.NewLine}";
496 
497  // Adicionar mensagem atual ao string builder, que produzirá
498  // todas as mensagens a mostrar de forma mais eficiente
499  sb.Append(lastMsg);
500  }
501 
502  // Remover última mensagem do string builder, pois esta será
503  // mostrada com cores diferentes
504  sb.Length = sb.Length - lastMsg.Length;
505 
506  // Posicionar o cursor para mostrar o painel de mensagens
507  SetCursor(0, posMessagesTop);
508 
509  // Definir cores das mensagens (exceto da última)
510  SetColor(colMessagesFg, colMessagesBg);
511 
512  // Mostrar mensagens (exceto a última)
513  // Notar que o string builder é automaticamente convertido em
514  // string através do seu método ToString()
515  Console.Write(sb);
516 
517  // Definir a cor para a última mensagem e mostrá-la
518  SetColor(colLastMessageFg, colLastMessageBg);
519  Console.Write(lastMsg);
520  }

625  public void RenderWorld(IReadOnlyWorld world)
626  {
627  // Percorrer linhas do mundo de simulação (até ao máximo calculado)
628  for (int y = 0; y < worldYRenderNCells; y++)
629  {
630  // Caso a cache ainda não tenha sido inicializada, colocar o
631  // cursor no início da linha atual
632  if (cache[0, y] == UNINITIALIZED)
633  SetCursor(posWorldLeft, posWorldTop + y);
634 
635  // Percorrer colunas do mundo para a linha atual (até ao máximo
636  // calculado)
637  for (int x = 0; x < worldXRenderNCells; x++)
638  {
639  // Obter objeto posição para os valores atuais de x e y
640  Coord coord = new Coord(x, y);
641 
642  // Está alguém na posição atual?
643  if (world.IsOccupied(coord))
644  {
645  // Em caso afirmativo, tratar de mostrar esse "alguém"
646 
647  // Variável que representa o ID único do agente
648  string agentID;
649 
650  // Agent na posição atual
651  Agent agent = world.GetAgentAt(coord);
652 
653  // Obter o ID único do agente
654  agentID = agent.ToString();
655 
656  // Se o ID único do agente tiver mais de 3 carateres,
657  // reduzir para 3 de modo a não desformatar mundo
658  if (agentID.Length > 3) agentID =
659  agentID.Substring(0, 3);
660 
661  // Este agente já cá estava no turno anterior? Se sim,
662  // não é necessário desenhá-lo novamente, pelo que
663  // podemos passar para a próxima iteração do ciclo
664  if (cache[x, y] == agentID) continue;
665 
666  // Caso a cache já tenha sido inicializada, colocar o
667  // cursos na posição correta (caso contrário já lá
668  // estaria)
669  if (cache[x, y] != UNINITIALIZED)
670  SetCursor(
671  posWorldLeft + worldCellLength * x,
672  posWorldTop + y);
673 
674  // Ativar cor correta para o agente atual
675  SetAgentColor(agent.Kind, agent.Movement);
676 
677  // Mostrar agente bem como o espaço que lhe se segue
678  Console.Write(agentID);
679  SetDefaultColor();
680  Console.Write(" ");
681 
682  // Atualizar cache com a informação sobre este agente
683  // nesta posição
684  cache[x, y] = agentID;
685  }
686  else
687  {
688  // Se chegarmos aqui é porque não está ninguém na
689  // posição atual
690 
691  // Se a cache indicar que não estava cá ninguém no
692  // turno anterior, escusamos de desenhar "..." e
693  // saltamos para a próxima iteração do ciclo
694  if (cache[x, y] == EMPTY) continue;
695 
696  // Caso a cache já tenha sido inicializada, colocar o
697  // cursos na posição correta (caso contrário já lá
698  // estaria)
699  if (cache[x, y] != UNINITIALIZED)
700  SetCursor(
701  posWorldLeft + worldCellLength * x,
702  posWorldTop + y);
703 
704  // Ativar cor por omissão e desenhar "...", que
705  // corresponde a um espaço vazio
706  SetDefaultColor();
707  Console.Write("... ");
708 
709  // Atualizar a cache sobre esta posição, nomeadamente
710  // com a indicação que não está cá ninguém
711  cache[x, y] = EMPTY;
712  }
713  }
714 
715  // É suposto desenhar o nevoeiro horizontal? Só o precisamos de
716  // fazer uma vez, antes do mundo ser renderizado pela primeira
717  // vez
718  if (worldXRenderFog && !worldRendered)
719  {
720  // Posicionar cursor no fim da linha atual
721  SetCursor(
722  posWorldLeft + worldCellLength * worldXRenderNCells,
723  posWorldTop + y);
724 
725  // Desenhar nevoeiro
726  Console.Write("~~~ ");
727  }
728  }
729 
730  // É suposto desenhar o nevoeiro vertical? Só o precisamos de fazer
731  // uma vez, antes do mundo ser renderizado pela primeira vez.
732  if (worldYRenderFog && !worldRendered)
733  {
734  // Vamos usar um string builder para construir uma linha
735  // inteira de nevoeiro, que será a última linha do mundo,
736  // contendo um nevoeiro para cada coluna
737  string fogLine = new StringBuilder().Insert(
738  // Inserir nevoeiro a partir da posição 0 do string builder
739  0, "~~~ ",
740  // Repetir esse nevoeiro o número de vezes necessária
741  worldXRenderNCells + (worldXRenderFog ? 1 : 0)).ToString();
742 
743  // Posicionar cursos no início da última linha
744  SetCursor(posWorldLeft, posWorldTop + worldYRenderNCells);
745 
746  // Desenhar nevoeiro guardado no string builder
747  Console.Write(fogLine);
748  }
749 
750  // Especificar que o mundo já foi renderizado uma vez
751  worldRendered = true;
752  }
