175  public static Options ParseArgs(string[] args)
176  {
177  // Opções a devolver (nota: as structs têm sempre um construtor
178  // vazio que inicializa todos os seus campos com o valor por
179  // omissão)
180  Options optionsObject = new Options();
181 
182  // Dicionário usado para ir guardando e processando as opções
183  IDictionary<string, uint> options = new Dictionary<string, uint>();
184 
185  // Se o número de opções não for válido...
186  if (args.Length != 2 * validOptions.Count)
187  {
188  // ...devolver um objeto Options com indicação desse erro
189  return new Options("Invalid number of arguments");
190  }
191 
192  // Processar as opções, "saltando" por cima dos valores das mesmas
193  // para já
194  for (int i = 0; i < args.Length; i += 2)
195  {
196  // Será a opção atual válida?
197  if (validOptions.Contains(args[i]))
198  {
199  // Será que esta opção ainda não foi indicada?
200  if (!options.ContainsKey(args[i]))
201  {
202  // Verificar se relativamente à opção atual foi dado
203  // um inteiro sem sinal válido
204  if (uint.TryParse(args[i + 1], out uint value))
205  {
206  // Pelos vistos sim, guardar esse valor no
207  // dicionário de opções
208  options[args[i]] = value;
209  }
210  else
211  {
212  // Não foi possível converter o valor num
213  // inteiro sem sinal, indicar esse erro e sair do
214  // ciclo
215  optionsObject = new Options(
216  $"Invalid value for option {args[i]}");
217  break;
218  }
219  }
220  else
221  {
222  // Opção já tinha sido indicada, indicar esse erro e
223  // sair do ciclo
224  optionsObject = new Options(
225  $"Repeated option: {args[i]}");
226  break;
227  }
228  }
229  else
230  {
231  // Opção desconhecida, indicar esse erro e sair do ciclo
232  optionsObject = new Options($"Unknown option: {args[i]}");
233  break;
234  }
235  }
236 
237  // Caso não tenha ocorrido nenhum erro...
238  if (!optionsObject.Error)
239  {
240  // ...construir instância de Options com os valores indicados
241  // pelo utilizador...
242  optionsObject = new Options(
243  options[validOptions[0]],
244  options[validOptions[1]],
245  options[validOptions[2]],
246  options[validOptions[3]],
247  options[validOptions[4]],
248  options[validOptions[5]],
249  options[validOptions[6]]);
250 
251  // ...e verificar se esses valores são válidos, isto é, se
252  // fazem sentido
253  optionsObject.Validate();
254 
255  }
256 
257  // Devolver objeto Options com as opções devidamente analisadas
258  return optionsObject;
259  }
