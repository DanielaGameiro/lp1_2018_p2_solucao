145  public void PlayTurn(out bool changePopulation, out string message)
146  {
147  // Obter coordenada de destino preferida, bem como uma mensagem
148  // sobre esse potencial movimento
149  Coord dest = moveBehavior.WhereToMove(this, out message);
150 
151  // Em princípio consideramos que não foram realizadas alterações
152  // à população de agentes
153  changePopulation = false;
154 
155  // O tipo de ação depende se a coordenada de destino está ocupada
156  // ou não por outro agente
157  if (!world.IsOccupied(dest))
158  {
159  // Senão estiver, simplesmente movemos o agente para lá...
160  world.MoveAgent(this, dest);
161  // ...atualizando a sua posição...
162  Pos = dest;
163  // ...bem como a mensagem, indicando que o movimento foi feito
164  // com sucesso.
165  message += " and succeeded";
166  }
167  else
168  {
169  // Se a coordenada estiver ocupada, obter referência ao agente
170  // que a está a ocupar
171  Agent other = world.GetAgentAt(dest);
172 
173  // É o próprio agente que está a ocupar a coordenada?
174  if (this == other)
175  {
176  // Nesse caso não é suposto o agente mover-se, pois não
177  // encontrou inimigos: basta completar a mensagem
178  message += " and failed to move";
179  }
180  else if (Kind == AgentKind.Zombie
181  && other.Kind == AgentKind.Human)
182  {
183  // Se o agente atual for um zombie e o agente que ocupa a
184  // coordenada de destino for um humano, então vamos infetar
185  // esse humando
186  world.GetAgentAt(dest).Infect();
187  // Significa que ocorreu uma modificação na população, pelo
188  // que temos de atualizar o respetivo parâmetro out...
189  changePopulation = true;
190  // ...e completar a mensagem em conformidade
191  message += $" and infected {other}";
192  }
193  else
194  {
195  // Caso contrário, simplesmente atualizamos a mensagem de
196  // modo a indicar que o agente não se moveu pois
197  // embateu noutro
198  message += $" but bumped into {other}";
199  }
200  }
201  }
