60  public override Coord WhereToMove(Agent agent, out string message)
61  {
62  // Agente alvo, a perseguir ou evitar
63  Agent target = null;
64 
65  // Variável que indica se o alvo/inimigo foi encontrado
66  bool foundEnemy = false;
67 
68  // Raio máximo de procura, igual à maior dimensão do mundo (x ou y)
69  // a dividir por 2
70  int maxRadius = Math.Max(world.XDim, world.YDim) / 2;
71 
72  // Vetor entre o agente atual e um possível alvo/inimigo
73  Coord vector = default(Coord);
74 
75  // Variável auxiliar para a procura de alvos/inimigos
76  Coord currentPos;
77 
78  // Percorrer os raios desde 1 até ao raio máximo
79  // Ciclo pode parar antecipadamente caso inimigo seja encontrado
80  for (int r = 1; r <= maxRadius && !foundEnemy; r++)
81  {
82  // Percorrer as coordenadas horizontais dentro do raio atual
83  for (int dx = -r; dx <= r && !foundEnemy; dx++)
84  {
85  // Percorrer as coordenadas verticais dentro do raio atual
86  for (int dy = -r; dy <= r && !foundEnemy; dy++)
87  {
88  // Caso a coordenada (x,y) atual não esteja no raio
89  // atual (porque está num raio menor), significa que já
90  // procuramos nesta posição e podemos passar para a
91  // próxima
92  if (Math.Max(Math.Abs(dx), Math.Abs(dy)) != r) continue;
93 
94  // Criar uma instância para a coordenada (x,y) atual
95  currentPos =
96  new Coord(agent.Pos.X + dx, agent.Pos.Y + dy);
97 
98  // Está alguém na coordenada atual?
99  if (world.IsOccupied(currentPos))
100  {
101  // Se sim, obter esse "alguém"
102  target = world.GetAgentAt(currentPos);
103 
104  // Esse "alguém" é inimigo?
105  if (target.Kind == enemy)
106  {
107  // Se sim, vetor de movimento entre agente a
108  // mover-se e o seu inimigo
109  vector = runAway
110  // Se for para fugir, vetor é entre inimigo
111  // a agente a mover-se
112  ? world.VectorBetween(currentPos, agent.Pos)
113  // Se for para perseguir, vetor é entre
114  // agente a mover-se e inimigo
115  : world.VectorBetween(agent.Pos, currentPos);
116 
117  // Indicar que foi encontrado inimigo de modo
118  // a sairmos dos ciclos
119  foundEnemy = true;
120  }
121  }
122  }
123  }
124  }
125 
126  // Inimigo foi encontrado?
127  if (foundEnemy)
128  {
129  // Se sim, atualizar mensagem com essa indicação...
130  string way = runAway ? "runaway from" : "move towards";
131  message = $"{agent} tried to {way} {target}";
132  // ...e devolver a posição vizinha determinada pela direção do
133  // vetor calculado anteriormente
134  return world.GetNeighbor(agent.Pos, vector);
135  }
136 
137  // Caso inimigo não tenha sido encontrado, atualizar mensagem com
138  // essa indicação e devolver posição onde o agente se encontra
139  // atualmente (indicando que não há movimento a ser feito)
140  message = $"{agent} didn't find enemies";
141  return agent.Pos;
142  }
143  }
