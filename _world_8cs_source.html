123  // Calcular distância horizontal direta entre as coordenadas, ou
124  // seja sem dar a volta pelos lados
125  xDistDirect = Math.Abs(c2.X - c1.X);
126 
127  // Calcular distância horizontal entre as coordenadas dando a volta

136  x += -Math.Sign(x) * XDim;
